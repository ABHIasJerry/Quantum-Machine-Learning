# ----- Python Support : >=3.14
# Setup the virtual environment

>> python -m venv .venv
>> .venv\Scripts\Activate
>> python.exe -m pip install --upgrade pip
>> pip install -r requirements.txt

# Run the code
Command line arguments : python main.py train
                         python main.py test

# Test model
>> Before testing, update the src/test/test.json value list with exact shape of sample data
[Refer from the prediction csv sample files for correct number of features and values.]

# Docker commands
# Build the image
docker build -t quantum-rfr-ml-abhi .

# Run the container
docker run --rm quantum-rfr-ml-abhi

# Or run interactively
docker run -it quantum-rfr-ml-abhi /bin/bash

# If script has input/output files
docker run --rm -v $(pwd):/app quantum-rfr-ml-abhi
